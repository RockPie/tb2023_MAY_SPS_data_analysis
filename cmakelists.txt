cmake_minimum_required(VERSION 3.12)
set(CMAKE_CXX_STANDARD 17)
project(sketchprj1)

find_package(ROOT REQUIRED COMPONENTS 
    Core 
    Tree
    Hist
    Gpad
    Graf3d
)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

add_library(sketchprj1 SHARED 
    src/SJ_datareading.cxx
    src/SJ_eventbuilder.cxx
    src/easylogging++.cc
    src/SJ_utilities.cxx
    src/SJ_plotting.cxx
    include/csv.h
    include/SJ_includes.h
)
target_link_libraries(sketchprj1 PUBLIC 
    ROOT::Core 
    ROOT::Tree
    ROOT::Hist
    # for TCanvas
    ROOT::Gpad
    ROOT::Graf3d
)

add_executable(Main Sketch1.cxx)
target_link_libraries(Main PUBLIC sketchprj1)
