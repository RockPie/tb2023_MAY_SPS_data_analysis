# * See https://root.cern/manual/integrate_root_into_my_cmake_project/ for more info
cmake_minimum_required(VERSION 3.12)
set(CMAKE_CXX_STANDARD 17)
project(sketchprj1)

find_package(ROOT REQUIRED COMPONENTS 
    Core 
    Tree
    Hist
    Gpad
    Graf3d
)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

add_library(sketchprj1_lib SHARED 
    src/SJ_datareading.cxx
    src/SJ_eventbuilder.cxx
    src/easylogging++.cc
    src/SJ_utilities.cxx
    src/SJ_plotting.cxx
    src/SJ_fitting.cxx
    include/csv.h
    include/SJ_includes.h
)

# * Change script name here to run different scripts
# add_executable(sketchprj1_main ./scripts/sum_analysis.cxx)
add_executable(sketchprj1_main ./scripts/Sketch1.cxx)

target_link_libraries(sketchprj1_lib PUBLIC 
    ROOT::Core 
    ROOT::Tree
    ROOT::Hist
    ROOT::Gpad
    ROOT::Graf3d
)

target_link_libraries(sketchprj1_main sketchprj1_lib)
